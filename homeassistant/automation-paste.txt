id: '1745593508763'
alias: Litter Robot Status Logger
description: 'Track visits and cycles with debounce to prevent sensor flutter overcounting'
triggers:
  - trigger: state
    entity_id: sensor.starshit_status_code
    from:
      - rdy
      - dfs
    to:
      - cd
      - cst
    id: visit_start
  - trigger: state
    entity_id: sensor.starshit_status_code
    from: ccp
    to:
      - rdy
      - dfs
    id: cycle_complete
conditions: []
actions:
  - choose:
      - conditions:
          - condition: trigger
            id: visit_start
          - condition: template
            value_template: >
              {% set last = states('input_datetime.litter_robot_last_visit') %}
              {{ last in ['unknown', 'unavailable', 'None', '']
                 or (now() - (last | as_datetime | as_local)).total_seconds() > 180 }}
        sequence:
          - action: counter.increment
            target:
              entity_id: counter.litter_robot_counter
          - action: input_datetime.set_datetime
            target:
              entity_id: input_datetime.litter_robot_last_visit
            data:
              datetime: '{{ now().strftime(''%Y-%m-%d %H:%M:%S'') }}'
      - conditions:
          - condition: trigger
            id: cycle_complete
        sequence:
          - action: counter.increment
            target:
              entity_id: counter.litter_robot_cycle_count
          - action: input_datetime.set_datetime
            target:
              entity_id: input_datetime.litter_robot_last_cycle
            data:
              timestamp: '{{ now().timestamp() }}'
  - action: rest_command.push_litterrobot_current
    data: {}
mode: single
